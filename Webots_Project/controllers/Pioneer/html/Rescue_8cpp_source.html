431 vector <float>Rescue:: RGB2HSV(unsigned char red,unsigned char green,unsigned char blue){
432  vector<float>HSV;
433  float R=(float)red;
434  float G=(float)green;
435  float B=(float)blue;
436  //Expresiones de conversión a HSV
437  HSV.push_back(180+((atan((sqrt(3)*(G-B))/((R-G)+(R-B)))*180)/M_PI));
438  HSV.push_back((R+G+B)/3);
439  HSV.push_back(1-(min({R,G,B})/HSV[1]));
440 
441  return HSV;
